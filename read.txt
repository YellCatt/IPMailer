cat > /etc/init.d/ipmailer << 'EOF'
#!/bin/sh /etc/rc.common

START=99
STOP=10

USE_PROCD=1

start_service() {
    procd_open_instance
    procd_set_param command /plugins/data/ipmailer/ipmailer
    procd_set_param respawn
    procd_set_param stdout 1
    procd_set_param stderr 1
    procd_close_instance
}

stop_service() {
    killall -q ipmailer 2>/dev/null
}
EOF

chmod +x /etc/init.d/ipmailer
/etc/init.d/ipmailer enable
/etc/init.d/ipmailer start
